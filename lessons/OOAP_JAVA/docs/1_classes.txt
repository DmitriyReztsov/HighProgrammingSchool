class Cell {
  int gold;
  int culture;
  int work;
  String relief;
  boolean isLocked;
  String resource;

  public Cell(int gold, int culture, int work, String relief, boolean isLocked, String resource) {
    this.gold = gold;
    this.culture = culture;
    this.work = work;
    this.relief = relief;
    this.isLocked = isLocked;
    this.resource = resource;
  }
}

class Unit {
  String name;
  int power;
  String type;  
  int movePoint;  
  int bonus;
  Cell currentCell;
  
  public Unit(String name, int power, String type, int movePoint, int bonus, Cell currentCell) {
    this.name = name;
    this.power = power;
    this.type = type;  
    this.movePoint = movePoint;  
    this.bonus = bonus;
    this.currentCell = currentCell;
  }
}

public class MainClass {
  public static void main(String[] args) {
    Cell cell1 = new Cell(
      1,
      1,
      1,
      "plain",
      false,
      null
    );
    Cell cell2 = new Cell(
      1,
      1,
      1,
      "forest",
      false,
      null
    );
    
    Unit unit1 = new Unit(
      "worker",
      0,
      "not-combat",
      5,
      0,
      cell1
    );
    Unit unit2 = unit1;  // создаем второго рабочего, копируем ссылку, а не значения
    
    // первый рабочий делает ход
    unit1.currentCell = cell2;
    
    // проверяем второго рабочего - а он тоже переместился!
    System.out.println("Second worker on cell " + unit2.currentCell.relief);
  }
}
