class FieldCell {
  protected Cell baseCell;
  protected CellConstruction construction;
  protected Unit occupiedByUnit;
  
  boolean isLocked;

  public FieldCell(Cell baseCell, CellConstruction construction, Unit occupiedByUnit) {
    this.baseCell = baseCell;
    this.construction = construction;
    this.occupiedByUnit = occupiedByUnit;
  }

  public int getDefensePoint() {
    return this.construction.getDefensePoint() + this.baseCell.getDefensePoint() + this.occupiedByUnit.getDefensePoint();
  }
}

class SwordsMan extends Unit {

    public SwordsMan() {
        super("swordsman");
    }

    public void attack(int damage) {
        System.out.println("Attack with sword. Damage: " + damage);
    }

    public void attack(int damage, int bonus) {
        System.out.println("Attack with bonus. Damage: " + (damage + bonus));
    }

    public void attack(String target) {
        System.out.println("Attack target: " + target);
    }
}

public class MainClass {
	
  public static void main(String[] args) {
    SwordsMan sm = new SwordsMan();
    sm.attack(50);
    sm.attack(75, 5);
    sm.attack("Dragon");
  }
